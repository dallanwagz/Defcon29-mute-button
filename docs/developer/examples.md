## Example 1: Blink LED 4 Three Times

```python
#!/usr/bin/env python3
"""Blink LED 4 red three times."""
import time
from dc29.badge import BadgeAPI
from dc29.protocol import MuteState

PORT = "/dev/tty.usbmodem14201"

badge = BadgeAPI(PORT)
for _ in range(3):
    badge.set_mute_state(MuteState.MUTED)    # red
    time.sleep(0.3)
    badge.set_mute_state(MuteState.NOT_IN_MEETING)  # off
    time.sleep(0.3)

badge.close()
print("Done")
```

## Example 2: Color Cycle All LEDs

```python
#!/usr/bin/env python3
"""Cycle all 4 LEDs through a color sequence."""
import time
from dc29.badge import BadgeAPI

PORT = "/dev/tty.usbmodem14201"

COLORS = [
    (255, 0, 0),    # red
    (0, 255, 0),    # green
    (0, 0, 255),    # blue
    (255, 165, 0),  # orange
    (160, 0, 255),  # purple
    (0, 200, 255),  # cyan
    (255, 255, 0),  # yellow
    (0, 0, 0),      # off
]

badge = BadgeAPI(PORT)
for r, g, b in COLORS:
    for led in (1, 2, 3, 4):
        badge.set_led(led, r, g, b)
    time.sleep(0.5)

badge.close()
print("Done")
```

## Example 3: Monitor Badge Events

```python
#!/usr/bin/env python3
"""Print all badge events as they arrive."""
import time
from dc29.badge import BadgeAPI
from dc29.protocol import modifier_name, keycode_name, EFFECT_NAMES

PORT = "/dev/tty.usbmodem14201"

badge = BadgeAPI(PORT)

badge.on_button_press = lambda btn, mod, kc: print(
    f"BUTTON {btn}: {modifier_name(mod)}+{keycode_name(kc, mod)}"
)
badge.on_effect_mode = lambda mode: print(
    f"EFFECT MODE: {EFFECT_NAMES.get(mode, mode)}"
)
badge.on_chord = lambda t: print(
    f"CHORD: {'long' if t == 2 else 'short'}"
)
badge.on_connect    = lambda: print("Badge connected")
badge.on_disconnect = lambda: print("Badge disconnected")

print(f"Monitoring {PORT} — press Ctrl+C to stop")
try:
    while True:
        time.sleep(1)
except KeyboardInterrupt:
    pass
finally:
    badge.close()
```

## Example 4: Set All Buttons to Custom Keymaps

```python
#!/usr/bin/env python3
"""Configure button keymaps for a macOS workflow."""
import time
from dc29.badge import BadgeAPI
from dc29.protocol import MOD_SHIFT_GUI, MOD_MEDIA

PORT = "/dev/tty.usbmodem14201"

# Keymaps: (button, modifier, keycode)
KEYMAPS = [
    (1, MOD_SHIFT_GUI, 0x10),   # button 1: Shift+Cmd+M (macOS Teams mute)
    (2, MOD_MEDIA,     0xE2),   # button 2: media mute
    (3, MOD_MEDIA,     0xE9),   # button 3: volume up
    (4, MOD_MEDIA,     0xEA),   # button 4: volume down
]

badge = BadgeAPI(PORT)

deadline = time.monotonic() + 5.0
while not badge.connected and time.monotonic() < deadline:
    time.sleep(0.05)

for button, mod, key in KEYMAPS:
    badge.set_key(button, mod, key)
    print(f"Set button {button}: mod=0x{mod:02X} key=0x{key:02X}")
    time.sleep(0.2)

badge.close()
print("All keymaps updated")
```

## Example 5: Asyncio Event Loop Integration

```python
#!/usr/bin/env python3
"""
Integrate BadgeAPI with asyncio.
Badge callbacks arrive on the reader thread; forward them to the event loop.
"""
import asyncio
from dc29.badge import BadgeAPI
from dc29.protocol import MuteState

PORT = "/dev/tty.usbmodem14201"


async def main() -> None:
    loop = asyncio.get_running_loop()
    event_queue: asyncio.Queue = asyncio.Queue()

    badge = BadgeAPI(PORT)

    # Wire callbacks: reader thread → safely enqueue into asyncio loop
    badge.on_button_press = lambda btn, mod, kc: loop.call_soon_threadsafe(
        event_queue.put_nowait, ("button", btn)
    )
    badge.on_effect_mode = lambda mode: loop.call_soon_threadsafe(
        event_queue.put_nowait, ("effect", mode)
    )
    badge.on_chord = lambda t: loop.call_soon_threadsafe(
        event_queue.put_nowait, ("chord", t)
    )

    print("Listening for badge events — press buttons or hold all 4 for a chord")
    print("Ctrl+C to stop")

    try:
        while True:
            event = await event_queue.get()
            kind = event[0]
            if kind == "button":
                btn = event[1]
                print(f"Button {btn} pressed")
                # Example: toggle LED 4 between red and green
                badge.set_mute_state(MuteState.MUTED)
                await asyncio.sleep(0.5)
                badge.set_mute_state(MuteState.UNMUTED)
            elif kind == "effect":
                print(f"Effect mode changed to: {event[1]}")
            elif kind == "chord":
                t = event[1]
                print(f"{'Long' if t == 2 else 'Short'} chord!")
    except KeyboardInterrupt:
        badge.set_mute_state(MuteState.NOT_IN_MEETING)
        print("\nExiting")
    finally:
        badge.close()


asyncio.run(main())
```

## Example 6: Build a Status Dashboard

```python
#!/usr/bin/env python3
"""
CPU usage indicator: green = low, yellow = medium, red = high.
LEDs 1-3 show CPU usage per core. LED 4 reserved.
Requires: pip install psutil
"""
import asyncio
import psutil
from dc29.badge import BadgeAPI

PORT = "/dev/tty.usbmodem14201"


def cpu_to_color(percent: float) -> tuple[int, int, int]:
    if percent < 33:
        return (0, 200, 0)    # green: low
    elif percent < 66:
        return (200, 150, 0)  # yellow: medium
    else:
        return (200, 0, 0)    # red: high


async def cpu_monitor(badge: BadgeAPI) -> None:
    while True:
        percents = psutil.cpu_percent(interval=1, percpu=True)
        for led_n, core_pct in enumerate(percents[:3], start=1):
            badge.set_led(led_n, *cpu_to_color(core_pct))


async def main() -> None:
    badge = BadgeAPI(PORT)
    print("CPU monitor running — LEDs 1-3 show core usage.  Ctrl-C to stop.")
    try:
        await cpu_monitor(badge)
    except KeyboardInterrupt:
        for n in range(1, 5):
            badge.set_led(n, 0, 0, 0)
        print("\nExiting")
    finally:
        badge.close()


asyncio.run(main())
```

## Example 7: Simple Toggle Script (No asyncio)

```python
#!/usr/bin/env python3
"""Toggle LED 4 between muted/unmuted on each run — useful as a keyboard shortcut."""
import sys
import time
from pathlib import Path
from dc29.badge import BadgeAPI
from dc29.protocol import MuteState

PORT = sys.argv[1] if len(sys.argv) > 1 else "/dev/tty.usbmodem14201"
STATE_FILE = Path.home() / ".dc29_mute_state"

current = STATE_FILE.read_text().strip() if STATE_FILE.exists() else "clear"
next_state = "muted" if current != "muted" else "unmuted"
mute_map = {
    "muted":   MuteState.MUTED,
    "unmuted": MuteState.UNMUTED,
    "clear":   MuteState.NOT_IN_MEETING,
}

badge = BadgeAPI(PORT)
deadline = time.monotonic() + 5.0
while not badge.connected and time.monotonic() < deadline:
    time.sleep(0.05)

badge.set_mute_state(mute_map[next_state])
time.sleep(0.1)
badge.close()

STATE_FILE.write_text(next_state)
print(f"LED 4: {current} → {next_state}")
```

Bind this to a keyboard shortcut to toggle the badge LED without running the full Teams bridge.

## Example 8: Minimal Custom Bridge

See [building-bridges.md](building-bridges.md) for the complete guide. Minimal runnable example:

```python
#!/usr/bin/env python3
"""A bridge that lights up 4 buttons and handles presses."""
import asyncio
from dc29.badge import BadgeAPI
from dc29.bridges.base import BaseBridge, BridgePage, PageButton


class DemoBridge(BaseBridge):

    @property
    def page(self) -> BridgePage:
        return BridgePage(
            name="demo",
            description="Demo bridge",
            buttons={
                1: PageButton("action-a", led=(200, 0, 0)),
                2: PageButton("action-b", led=(0, 60, 180)),
                3: PageButton("action-c", led=(180, 160, 0)),
                4: PageButton("action-d", led=(0, 180, 0)),
            },
        )

    async def handle_button(self, btn: int) -> None:
        labels = {1: "A", 2: "B", 3: "C", 4: "D"}
        print(f"Button {btn} pressed → action-{labels.get(btn, '?')}")

    async def run(self) -> None:
        self._install_button_hook()
        self._apply_page_leds()
        print("Demo bridge running — press buttons. Ctrl-C to stop.")
        try:
            while True:
                await asyncio.sleep(3600)
        except asyncio.CancelledError:
            pass
        finally:
            self._uninstall_button_hook()
            self._clear_page_leds()


badge = BadgeAPI("/dev/tty.usbmodem14201")
bridge = DemoBridge(badge)
try:
    asyncio.run(bridge.run())
except KeyboardInterrupt:
    pass
finally:
    badge.close()
```
