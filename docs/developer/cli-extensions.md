## Adding a Simple Command

```python
# dc29/cli.py
from typing import Optional
import typer
from dc29.protocol import EffectMode

@app.command()
def breathe(
    port: Optional[str] = typer.Option(None, "--port", "-p", envvar="DC29_PORT"),
) -> None:
    """Set LED effect to breathe mode."""
    from dc29.badge import BadgeAPI
    badge = BadgeAPI(_resolve_port(port))
    badge.set_effect_mode(EffectMode.BREATHE)
    typer.echo("Effect set to breathe")
    badge.close()
```

Typer automatically generates `--help` from the docstring and validates typed arguments.

## Adding a Command with Arguments

```python
@app.command()
def flash(
    port: Optional[str] = typer.Option(None, "--port", "-p", envvar="DC29_PORT"),
    n: int = typer.Argument(help="LED number 1-4"),
    r: int = typer.Argument(help="Red 0-255"),
    g: int = typer.Argument(help="Green 0-255"),
    b: int = typer.Argument(help="Blue 0-255"),
) -> None:
    """Flash an LED with a specific color three times."""
    if not (1 <= n <= 4):
        typer.echo(f"LED must be 1-4, got {n}", err=True)
        raise typer.Exit(1)
    import time
    from dc29.badge import BadgeAPI
    badge = BadgeAPI(_resolve_port(port))
    for _ in range(3):
        badge.set_led(n, r, g, b)
        time.sleep(0.2)
        badge.set_led(n, 0, 0, 0)
        time.sleep(0.2)
    badge.close()
    typer.echo(f"Flashed LED {n}")
```

## Adding a Command That Reads a Badge Reply

For commands that wait for a response (like `get-key`), register a callback and use a threading event:

```python
import threading
import time

@app.command("get-key")
def get_key(
    button: int = typer.Argument(help="Button 1-4"),
    port: Optional[str] = typer.Option(None, "--port", "-p", envvar="DC29_PORT"),
) -> None:
    """Query a button's current keymap."""
    from dc29.badge import BadgeAPI
    from dc29.protocol import modifier_name, keycode_name

    result = {}
    done = threading.Event()

    def _on_reply(n: int, mod: int, kc: int) -> None:
        result["n"], result["mod"], result["kc"] = n, mod, kc
        done.set()

    badge = BadgeAPI(_resolve_port(port))
    badge.on_key_reply = _on_reply

    deadline = time.monotonic() + 5.0
    while not badge.connected and time.monotonic() < deadline:
        time.sleep(0.05)

    badge.query_key(button)

    if done.wait(timeout=3.0):
        mod, kc = result["mod"], result["kc"]
        typer.echo(
            f"Button {button}: {modifier_name(mod)}+{keycode_name(kc, mod)}"
            f"  (mod=0x{mod:02X} key=0x{kc:02X})"
        )
    else:
        typer.echo("Timeout: no reply from badge", err=True)
        raise typer.Exit(1)

    badge.close()
```

## Adding a Bridge Command

Bridge commands run an asyncio event loop:

```python
@app.command()
def myservice(
    port: Optional[str] = typer.Option(None, "--port", "-p", envvar="DC29_PORT"),
    verbose: bool = typer.Option(False, "--verbose", "-v"),
) -> None:
    """Run the MyService bridge."""
    import asyncio
    import logging
    logging.basicConfig(
        level=logging.DEBUG if verbose else logging.INFO,
        format="%(asctime)s %(levelname)s %(message)s",
        datefmt="%H:%M:%S",
    )
    from dc29.badge import BadgeAPI
    from dc29.bridges.myservice import MyServiceBridge
    from dc29.config import get_config

    cfg = get_config()
    badge = BadgeAPI(_resolve_port(port or cfg.badge_port), brightness=cfg.badge_brightness)
    try:
        asyncio.run(MyServiceBridge(badge).run())
    except KeyboardInterrupt:
        pass
    finally:
        badge.close()
```

## Adding a Subgroup

Group related commands under a namespace:

```python
led_app = typer.Typer(name="led", help="LED control commands")
app.add_typer(led_app)

@led_app.command("set")
def led_set(
    port: Optional[str] = typer.Option(None, "--port", "-p", envvar="DC29_PORT"),
    n: int = typer.Argument(help="LED number 1-4"),
    color: str = typer.Argument(help="Color name or r,g,b"),
) -> None:
    """Set an LED color."""
    from dc29.badge import BadgeAPI
    from dc29.protocol import parse_color
    r, g, b = parse_color(color)
    badge = BadgeAPI(_resolve_port(port))
    badge.set_led(n, r, g, b)
    import time; time.sleep(0.1)
    badge.close()
```

Users call: `dc29 led set 1 red`

## Output Formatting with Rich

`rich` is already a dependency. Use it for tables:

```python
from rich.console import Console
from rich.table import Table

@app.command()
def my_table(port: Optional[str] = typer.Option(None, "--port", "-p", envvar="DC29_PORT")) -> None:
    """Show a status table."""
    console = Console()
    table = Table(title="DC29 Status")
    table.add_column("Item", style="cyan")
    table.add_column("Value", style="green")
    table.add_row("Port", _resolve_port(port))
    console.print(table)
```

## Environment Variable

All built-in commands check `DC29_PORT` as a fallback for `--port`. Set it once and omit `--port` from every command:

```bash
export DC29_PORT=/dev/tty.usbmodem14201

dc29 info
dc29 flow -v
dc29 set-key 1 gui+shift m
```
